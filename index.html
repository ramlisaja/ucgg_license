async function createLicense() { const expiryDays = parseInt(document.getElementById('expiryDays').value) || 30; const plan = document.getElementById('plan').value; const note = document.getElementById('note').value.trim(); try { const res = await fetch(`${API_URL}/licenses`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${ADMIN_SECRET}` }, body: JSON.stringify({ expiryDays, plan, note }) }); const data = await res.json(); console.log('Response:', data); if (data.success) { showToast(`✅ License created: ${data.license.key}`, 'success'); document.getElementById('note').value = ''; refreshData(); } else { showToast('❌ Failed to create license: ' + (data.error || 'Unknown error'), 'error'); } } catch (error) { console.error('Error:', error); showToast('❌ Failed to create license: ' + error.message, 'error'); } }
